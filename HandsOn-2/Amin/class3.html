f
f['last_pollution']=f['pollution'].shift(1)
f['next_pollution']=f['pollution'].shift(-1)
#f['']
f['rain_in_last24hours']=(f['percipitation'].rolling(8).sum()>0.0001).astype('int')
f['rain_in_last3hours'] =(f['percipitation'].rolling(2).sum()>0.0001).astype('int')

f['xd3']=np.sin(f['az']*np.pi/180)*f['windspeed']*3
f['yd3']=np.cos(f['az']*np.pi/180)*f['windspeed']*3


f['xd24']=f['xd3'].rolling(8).sum()
f['yd24']=f['xd3'].rolling(8).sum()
f['d24']=np.sqrt(f['xd24']**2+f['yd24']**2)

f['xd48']=f['xd3'].rolling(16).sum()
f['yd48']=f['xd3'].rolling(16).sum()
f['d48']=np.sqrt(f['xd48']**2+f['yd48']**2)
